  <script>

    const preguntas = [

      "¿Te has sentido nervioso, ansioso o al borde?",

      "¿Te has sentido incapaz de parar de preocuparte?",

      "¿Te has preocupado demasiado por diferentes cosas?",

      "¿Has tenido dificultad para relajarte?",

      "¿Te has sentido tan inquieto que te cuesta quedarte quieto?",

      "¿Te has irritado o sentido fácilmente molesto?",

      "¿Has sentido miedo como si algo terrible pudiera pasar?"

    ];

 

    const respuestas = ["Nunca (0)", "Varios días (1)", "Más de la mitad de los días (2)", "Casi todos los días (3)"];

 

    let resultadoHTML = "<form id='form'>";

    preguntas.forEach((pregunta, index) => {

      resultadoHTML += `<div class='question'><p>${index + 1}. ${pregunta}</p>`;

      respuestas.forEach((res, i) => {

        resultadoHTML += `<label><input type='radio' name='q${index}' value='${i}' required> ${res}</label>`;

      });

      resultadoHTML += `</div>`;

    });

    resultadoHTML += `<button type='submit'>Ver resultado</button></form>`;

    resultadoHTML += `<div id='resultado' class='result'></div>`;

 

    document.getElementById("test").innerHTML = resultadoHTML;

 

    document.getElementById("form").onsubmit = function(e) {

      e.preventDefault();

      let total = 0;

      for (let i = 0; i < 7; i++) {

        const val = document.querySelector(`input[name='q${i}']:checked`).value;

        total += parseInt(val);

      }

      let mensaje = "";

      if (total <= 4) mensaje = "Nivel mínimo de ansiedad. ¡Sigue cuidando tu salud mental!";

      else if (total <= 9) mensaje = "Ansiedad leve. Considera hablar con un profesional.";

      else if (total <= 14) mensaje = "Ansiedad moderada. Te recomendamos apoyo psicológico.";

      else mensaje = "Ansiedad severa. Agenda una consulta profesional lo antes posible.";

 

      document.getElementById("resultado").innerHTML = `Tu puntaje es <strong>${total}</strong>. ${mensaje}

      <div class='cta'>

        <a href='https://wa.me/573003003504' target='_blank'>📲 Agenda por WhatsApp 300 3003504</a>

        <a href='https://wa.me/573192288288' target='_blank'>📲 Agenda por WhatsApp 319 2288288</a>

        <a href='https://www.globalsaludmental.com' target='_blank'>🌐 Visita nuestro sitio web</a>

      </div>`;

    }

  </script>
